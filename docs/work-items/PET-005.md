#### 1. Rust 校验模块

创建 `src-tauri/src/skin_validator.rs`：

```rust
use std::path::{Path, PathBuf};
use std::collections::HashSet;

pub const REQUIRED_STATES: [&str; 11] = [
    "idle", "walk", "click", "drag", "sleep", "wake",
    "hover", "happy", "sad", "surprised", "wave"
];

#[derive(Debug, Clone)]
pub struct ValidationError {
    pub kind: String,
    pub message: String,
}

#[derive(Debug, Clone)]
pub struct ValidationResult {
    pub valid: bool,
    pub errors: Vec<ValidationError>,
}

pub fn validate_skin(skin_dir: &Path, skin: &SkinConfig) -> ValidationResult {
    let mut errors = Vec::new();
    
    // 1. 检查 coreProfile
    if skin.core_profile != "rich-v1" {
        errors.push(ValidationError {
            kind: "profile".into(),
            message: format!("Expected coreProfile 'rich-v1', got '{}'", skin.core_profile),
        });
    }
    
    // 2. 检查强制键
    let skin_states: HashSet<&str> = skin.states.keys().map(|s| s.as_str()).collect();
    for required in REQUIRED_STATES.iter() {
        if !skin_states.contains(required) {
            errors.push(ValidationError {
                kind: "missing_state".into(),
                message: format!("Missing required state: {}", required),
            });
        }
    }
    
    // 3. 检查帧文件存在
    for (state_name, state) in &skin.states {
        for frame in &state.frames {
            let frame_path = skin_dir.join(frame);
            if !frame_path.exists() {
                errors.push(ValidationError {
                    kind: "missing_frame".into(),
                    message: format!("State '{}' frame '{}' not found", state_name, frame),
                });
            }
        }
    }
    
    // 4. 检查 PNG 格式（简单检查：文件头）
    for (state_name, state) in &skin.states {
        for frame in &state.frames {
            let frame_path = skin_dir.join(frame);
            if frame_path.exists() {
                if let Ok(bytes) = std::fs::read(&frame_path) {
                    // PNG 文件头: 89 50 4E 47 0D 0A 1A 0A
                    if bytes.len() < 8 || bytes[0..8] != [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A] {
                        errors.push(ValidationError {
                            kind: "invalid_png".into(),
                            message: format!("State '{}' frame '{}' is not a valid PNG", state_name, frame),
                        });
                    }
                }
            }
        }
    }
    
    // 5. 检查帧尺寸（需要读取 PNG IHDR，简化版：跳过或用 image crate）
    // 可选：使用 `image` crate 读取尺寸
    // 当前简化版：仅检查文件存在
    
    ValidationResult {
        valid: errors.is_empty(),
        errors,
    }
}
```

```rust
use crate::skin_validator::{validate_skin, ValidationResult};

#[tauri::command]
pub fn validate_skin_path(app: tauri::AppHandle, path: String) -> Result<ValidationResult, String> {
    let resource_path = app.path().resource_dir()
        .map_err(|e| e.to_string())?;
    let skin_dir = safe_path_join(&resource_path, &path)?;
    
    // 先加载 skin.json
    let skin_json_path = skin_dir.join("skin.json");
    let content = std::fs::read_to_string(skin_json_path)
        .map_err(|e| format!("Failed to read skin.json: {}", e))?;
    let skin: SkinConfig = serde_json::from_str(&content)
        .map_err(|e| format!("Failed to parse skin.json: {}", e))?;
    
    Ok(validate_skin(&skin_dir, &skin))
}
```

```rust
#[tauri::command]
async fn load_skin_safe(app: tauri::AppHandle, path: String) -> Result<SkinConfig, String> {
    // 尝试加载指定皮肤
    let result = load_skin_internal(&app, &path);
    
    match result {
        Ok(skin) => {
            // 校验
            let validation = validate_skin_path(app.clone(), path.clone())?;
            if validation.valid {
                Ok(skin)
            } else {
                // 校验失败，回退默认
                log::warn!("Skin validation failed: {:?}", validation.errors);
                load_skin_internal(&app, "assets/skins/default")
                    .map_err(|e| format!("Fallback failed: {}", e))
            }
        }
        Err(e) => {
            // 加载失败，回退默认
            log::warn!("Skin load failed: {}, falling back to default", e);
            load_skin_internal(&app, "assets/skins/default")
                .map_err(|e| format!("Fallback failed: {}", e))
        }
    }
}
```
