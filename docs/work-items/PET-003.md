#### 1. Rust 状态机模块

```rust
// src-tauri/src/state_machine.rs
use std::sync::Mutex;
use tauri::State;

pub struct PetStateMachine {
    current_state: String,
    previous_state: String,
    frame_index: usize,
}

impl PetStateMachine {
    pub fn new() -> Self {
        Self {
            current_state: "idle".to_string(),
            previous_state: "idle".to_string(),
            frame_index: 0,
        }
    }

    pub fn switch_state(&mut self, new_state: String) {
        self.previous_state = self.current_state.clone();
        self.current_state = new_state;
        self.frame_index = 0;
    }

    pub fn current_state(&self) -> &str {
        &self.current_state
    }

    pub fn advance_frame(&mut self, total_frames: usize) -> bool {
        self.frame_index += 1;
        if self.frame_index >= total_frames {
            self.frame_index = 0;
            return true; // 循环完成
        }
        false
    }
}
```

#### 2. Tauri 命令

```rust
#[tauri::command]
fn switch_state(state_machine: State<Mutex<PetStateMachine>>, new_state: String) -> Result<(), String> {
    let mut sm = state_machine.lock().map_err(|e| e.to_string())?;
    sm.switch_state(new_state);
    Ok(())
}

#[tauri::command]
fn get_current_frame(state_machine: State<Mutex<PetStateMachine>>) -> Result<(String, usize), String> {
    let sm = state_machine.lock().map_err(|e| e.to_string())?;
    Ok((sm.current_state().to_string(), sm.frame_index))
}
```

#### 3. 路径安全修复（TECH_DEBT #1）

在 `commands.rs` 中添加：

```rust
use std::path::{Path, PathBuf};

fn safe_path_join(base: &Path, user_input: &str) -> Result<PathBuf, String> {
    // 禁止路径遍历
    if user_input.contains("..") || user_input.contains('\\') {
        return Err("Invalid path: traversal attempt".into());
    }
    let joined = base.join(user_input);
    // 规范化并验证仍在 base 内
    let canonical_base = base.canonicalize().map_err(|e| format!("Base path error: {}", e))?;
    let canonical_joined = joined.canonicalize().map_err(|e| format!("Joined path error: {}", e))?;
    if !canonical_joined.starts_with(&canonical_base) {
        return Err("Path traversal detected".into());
    }
    Ok(canonical_joined)
}
```

#### 4. 前端帧序列播放

```tsx
// src/components/Pet.tsx
import { useEffect, useState, useRef } from 'react';
import { invoke } from '@tauri-apps/api/core';

interface State {
  frames: string[];
  fps: number;
  loopType: 'loop' | 'oneshot';
}

interface SkinConfig {
  id: string;
  name: string;
  states: Record<string, State>;
  canvas: { width: number; height: number };
}

export function Pet() {
  const [skin, setSkin] = useState<SkinConfig | null>(null);
  const [currentState, setCurrentState] = useState<string>('idle');
  const [frameIndex, setFrameIndex] = useState<number>(0);
  const [frameBase64, setFrameBase64] = useState<string>('');
  const animationRef = useRef<number | null>(null);
  const skinPath = 'assets/skins/default';

  // 加载皮肤
  useEffect(() => {
    invoke<SkinConfig>('load_skin', { path: skinPath }).then(setSkin).catch(console.error);
  }, []);

  // 播放帧序列
  useEffect(() => {
    if (!skin || !skin.states[currentState]) return;
    
    const state = skin.states[currentState];
    const fps = state.fps || 1;
    const frameDelay = 1000 / fps;
    
    const playFrame = async () => {
      const frameName = state.frames[frameIndex];
      const base64 = await invoke<string>('get_frame_base64', {
        skinPath,
        frameName
      });
      setFrameBase64(`data:image/png;base64,${base64}`);
      
      // 下一帧
      const nextIndex = frameIndex + 1;
      if (nextIndex >= state.frames.length) {
        if (state.loopType === 'oneshot') {
          // oneshot 结束后回到 idle
          invoke('switch_state', { newState: 'idle' }).catch(console.error);
          setCurrentState('idle');
          setFrameIndex(0);
        } else {
          // loop 循环
          setFrameIndex(0);
        }
      } else {
        setFrameIndex(nextIndex);
      }
    };
    
    playFrame();
    animationRef.current = window.setInterval(playFrame, frameDelay);
    
    return () => {
      if (animationRef.current) clearInterval(animationRef.current);
    };
  }, [skin, currentState, frameIndex]);

  // 定时进入 sleep（idle 30s）
  useEffect(() => {
    if (currentState !== 'idle') return;
    
    const timeout = setTimeout(() => {
      invoke('switch_state', { newState: 'sleep' }).catch(console.error);
      setCurrentState('sleep');
      setFrameIndex(0);
    }, 30000);
    
    return () => clearTimeout(timeout);
  }, [currentState]);

  return (
    <div 
      style={{ 
        width: skin?.canvas.width || 512, 
        height: skin?.canvas.height || 512 
      }}
      onMouseMove={(e) => {
        // 鼠标移动唤醒 sleep
        if (currentState === 'sleep') {
          invoke('switch_state', { newState: 'wake' }).catch(console.error);
          setCurrentState('wake');
          setFrameIndex(0);
        }
      }}
    >
      {frameBase64 && (
        <img 
          src={frameBase64} 
          alt="pet" 
          style={{ width: '100%', height: '100%', objectFit: 'contain' }}
        />
      )}
    </div>
  );
}
```
