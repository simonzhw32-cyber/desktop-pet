```rust
use tauri::{Manager, Window};

#[tauri::command]
pub fn set_ignore_cursor_events(window: Window, ignore: bool) -> Result<(), String> {
    window.set_ignore_cursor_events(ignore).map_err(|e| e.to_string())
}

#[tauri::command]
pub fn start_dragging(window: Window) -> Result<(), String> {
    // 开始拖拽时禁用 click-through
    window.set_ignore_cursor_events(false)?;
    // 使用 Tauri 内置拖拽 API
    window.start_dragging().map_err(|e| e.to_string())
}

#[tauri::command]
pub fn stop_dragging(window: Window) -> Result<(), String> {
    // 恢复 click-through
    window.set_ignore_cursor_events(true).map_err(|e| e.to_string())
}
```

#### 4. 前端鼠标交互

重写 `src/components/Pet.tsx`（含帧堆积修复）：

```tsx
import { useEffect, useState, useRef, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';
import { getCurrentWindow } from '@tauri-apps/api/window';

// ... SkinConfig 类型定义

const SKIN_PATH = 'assets/skins/default';

export function Pet() {
  const [skin, setSkin] = useState<SkinConfig | null>(null);
  const [frameBase64, setFrameBase64] = useState<string>('');
  const [currentState, setCurrentState] = useState<string>('idle');
  const frameIndexRef = useRef<number>(0);
  const isDraggingRef = useRef<boolean>(false);
  const hoverTimeoutRef = useRef<number | null>(null);

  // 加载皮肤（保持不变）

  // 帧播放（修复帧堆积）
  const playNextFrame = useCallback(async () => {
    if (!skin || !skin.states[currentState]) return;
    
    const state = skin.states[currentState];
    const totalFrames = state.frames.length;
    const frameName = state.frames[frameIndexRef.current];
    
    try {
      const base64 = await invoke<string>('get_frame_base64', {
        skinPath: SKIN_PATH,
        frameName
      });
      setFrameBase64(`data:image/png;base64,${base64}`);
      
      // 推进帧索引
      frameIndexRef.current++;
      if (frameIndexRef.current >= totalFrames) {
        if (state.loopType === 'oneshot') {
          setCurrentState('idle');
          frameIndexRef.current = 0;
        } else {
          frameIndexRef.current = 0;
        }
      }
      
      // 用 setTimeout 递归调度下一帧（避免帧堆积）
      const fps = state.fps || 1;
      setTimeout(playNextFrame, 1000 / fps);
    } catch (e) {
      console.error('Frame load error:', e);
    }
  }, [skin, currentState]);

  useEffect(() => {
    if (!skin) return;
    frameIndexRef.current = 0;
    playNextFrame();
  }, [skin, currentState, playNextFrame]);

  // idle → sleep 定时器（保持不变）

  // 鼠标事件处理
  const handleMouseEnter = () => {
    if (currentState !== 'idle' || isDraggingRef.current) return;
    
    // 延迟 200ms 触发 hover（避免划过就触发）
    hoverTimeoutRef.current = window.setTimeout(() => {
      setCurrentState('hover');
      frameIndexRef.current = 0;
    }, 200);
  };

  const handleMouseLeave = () => {
    if (hoverTimeoutRef.current) {
      clearTimeout(hoverTimeoutRef.current);
    }
    if (currentState === 'hover') {
      setCurrentState('idle');
      frameIndexRef.current = 0;
    }
  };

  const handleClick = () => {
    if (isDraggingRef.current) return;
    setCurrentState('click');
    frameIndexRef.current = 0;
  };

  const handleMouseDown = async (e: React.MouseEvent) => {
    if (e.button !== 0) return; // 仅左键
    isDraggingRef.current = true;
    await invoke('start_dragging');
  };

  const handleMouseUp = async () => {
    if (!isDraggingRef.current) return;
    isDraggingRef.current = false;
    await invoke('stop_dragging');
    setCurrentState('idle');
  };

  useEffect(() => {
    const handleGlobalMouseUp = () => {
      if (isDraggingRef.current) {
        isDraggingRef.current = false;
        invoke('stop_dragging').catch(console.error);
      }
    };
    
    window.addEventListener('mouseup', handleGlobalMouseUp);
    return () => window.removeEventListener('mouseup', handleGlobalMouseUp);
  }, []);

  // sleep → wake（保持不变）

  return (
    <div
      style={{ width: skin?.canvas.width || 512, height: skin?.canvas.height || 512 }}
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
      onClick={handleClick}
      onMouseDown={handleMouseDown}
      onMouseUp={handleMouseUp}
    >
      <img src={frameBase64} alt="pet" style={{ width: '100%', height: '100%', objectFit: 'contain' }} />
    </div>
  );
}
```
